using System.Runtime.ExceptionServices;

namespace LabTask_1
{
    internal class Program
    {
        static void Main(string[] args)    //Main Function
        {
            Circle();
            Rectangule();
            Triangle();
        }

        static void Circle()  // Area of circle
        {
            Console.Write("Enter Radius: ");
            double Radious = Convert.ToDouble(Console.ReadLine());
            
            double Area = Math.PI * Radious * Radious;
            Console.WriteLine("Area of circle: " + Area);
        }

        static void Rectangule()  // Area of rectangule
        {
            Console.Write("Enter Length: ");
            double Length = Convert.ToDouble(Console.ReadLine());
            Console.Write("Enter Width: ");
            double Width = Convert.ToDouble(Console.ReadLine());
            
            double Area = Length * Width;
            Console.WriteLine("Area of  rectangule: " + Area);
        }


        static void Triangle()  // Area of Triangle
        {
            Console.Write("Enter Base: ");
            double Base = Convert.ToDouble(Console.ReadLine());
            Console.Write("Enter Hight: ");
            double Hight = Convert.ToDouble(Console.ReadLine());
            
            double Area = (Base*Hight)/2;
            Console.WriteLine("Area of  triangle: " + Area);
        }
    }
}

{
    class Program
    {
        static void Main(string[] args)
        {
              for (int column = 0; column <= 8; column++)
            {
                for (int row = 0; row <= 8; row++)
                {
                    if ((column + row) % 2 == 0)
                    {
                        Console.Write("X");
                    }
                    else if ((column + row) % 2 == 1)
                    {
                        Console.Write("O");
                    }                  
                }
               Console.WriteLine();
            }
        }
    }
